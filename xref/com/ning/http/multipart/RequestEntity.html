26  public interface RequestEntity {

28      /**
29       * Writes the request entity to the given stream.
30       *
31       * @param out
32       * @throws java.io.IOException
33       */
34      void writeRequest(OutputStream out) throws IOException;
35  
36      /**
37       * Gets the request entity's length. This method should return a non-negative value if the content

39       * EntityEnclosingMethod will use chunk encoding to
40       * transmit the request entity.
41       *
42       * @return a non-negative value when content length is known or a negative value when content length
43       *         is not known
44       */
45      long getContentLength();
46  
47      /**
48       * Gets the entity's content type.  This content type will be used as the value for the
49       * "Content-Type" header.
50       *
51       * @return the entity's content type
52       */
53      String getContentType();

55  }
