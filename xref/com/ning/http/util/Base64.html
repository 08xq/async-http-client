13  package com.ning.http.util;

15  /**
16   * Implements the "base64" binary encoding scheme as defined by
17   * <a href="http://tools.ietf.org/html/rfc2045">RFC 2045</a>.
18   * <p/>
19   * Portions of code here are taken from Apache Pivot
20   */
21  public final class Base64 {
22      private static final char[] lookup = new char[64];
23      private static final byte[] reverseLookup = new byte[256];
24  
25      static {
26          // Populate the lookup array
27  
28          for (int i = 0; i < 26; i++) {
29              lookup[i] = (char) ('A' + i);
30          }
31  
32          for (int i = 26, j = 0; i < 52; i++, j++) {
33              lookup[i] = (char) ('a' + j);
34          }
35  
36          for (int i = 52, j = 0; i < 62; i++, j++) {
37              lookup[i] = (char) ('0' + j);
38          }
39  
40          lookup[62] = '+';
41          lookup[63] = '/';
42  
43          // Populate the reverse lookup array
44  
45          for (int i = 0; i < 256; i++) {
46              reverseLookup[i] = -1;
47          }
48  
49          for (int i = 'Z'; i >= 'A'; i--) {
50              reverseLookup[i] = (byte) (i - 'A');
51          }
52  
53          for (int i = 'z'; i >= 'a'; i--) {
54              reverseLookup[i] = (byte) (i - 'a' + 26);
55          }
56  
57          for (int i = '9'; i >= '0'; i--) {
58              reverseLookup[i] = (byte) (i - '0' + 52);
59          }
60  
61          reverseLookup['+'] = 62;
62          reverseLookup['/'] = 63;
63          reverseLookup['='] = 0;
64      }
65  
66      /**
67       * This class is not instantiable.
68       */
69      private Base64() {
70      }
71  
72      /**
73       * Encodes the specified data into a base64 string.
74       *
75       * @param bytes The unencoded raw data.
76       */
77      public static String encode(byte[] bytes) {
78          // always sequence of 4 characters for each 3 bytes; padded with '='s as necessary:
79          StringBuilder buf = new StringBuilder(((bytes.length + 2) / 3) * 4);
80  
81          // first, handle complete chunks (fast loop)
82          int i = 0;
83          for (int end = bytes.length - 2; i < end; ) {
84              int chunk = ((bytes[i++] & 0xFF) << 16)
85                      | ((bytes[i++] & 0xFF) << 8)
86                      | (bytes[i++] & 0xFF);
87              buf.append(lookup[chunk >> 18]);
88              buf.append(lookup[(chunk >> 12) & 0x3F]);
89              buf.append(lookup[(chunk >> 6) & 0x3F]);
90              buf.append(lookup[chunk & 0x3F]);
91          }
92  
93          // then leftovers, if any
94          int len = bytes.length;
95          if (i < len) { // 1 or 2 extra bytes?
96              int chunk = ((bytes[i++] & 0xFF) << 16);
97              buf.append(lookup[chunk >> 18]);
98              if (i < len) { // 2 bytes
99                  chunk |= ((bytes[i] & 0xFF) << 8);
100                 buf.append(lookup[(chunk >> 12) & 0x3F]);
101                 buf.append(lookup[(chunk >> 6) & 0x3F]);
102             } else { // 1 byte
103                 buf.append(lookup[(chunk >> 12) & 0x3F]);
104                 buf.append('=');
105             }
106             buf.append('=');
107         }
108         return buf.toString();
109     }

111     /**
112      * Decodes the specified base64 string back into its raw data.
113      *
114      * @param encoded The base64 encoded string.
115      */
116     public static byte[] decode(String encoded) {
117         int padding = 0;
118 
119         for (int i = encoded.length() - 1; encoded.charAt(i) == '='; i--) {
120             padding++;
121         }
122 
123         int length = encoded.length() * 6 / 8 - padding;
124         byte[] bytes = new byte[length];
125 
126         for (int i = 0, index = 0, n = encoded.length(); i < n; i += 4) {
127             int word = reverseLookup[encoded.charAt(i)] << 18;
128             word += reverseLookup[encoded.charAt(i + 1)] << 12;
129             word += reverseLookup[encoded.charAt(i + 2)] << 6;
130             word += reverseLookup[encoded.charAt(i + 3)];
131 
132             for (int j = 0; j < 3 && index + j < length; j++) {
133                 bytes[index + j] = (byte) (word >> (8 * (2 - j)));
134             }
135 
136             index += 3;
137         }
138 
139         return bytes;
140     }
