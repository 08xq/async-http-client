45  import java.text.ParseException;
46  import java.text.SimpleDateFormat;
47  import java.util.Arrays;
48  import java.util.Calendar;
49  import java.util.Collection;
50  import java.util.Date;
51  import java.util.Iterator;
52  import java.util.Locale;
53  import java.util.TimeZone;

55  /**
56   * A utility class for parsing and formatting HTTP dates as used in cookies and
57   * other headers.  This class handles dates as defined by RFC 2616 section
58   * 3.3.1 as well as some other common non-standard formats.
59   *
60   * @author Christopher Brown
61   * @author Michael Becke
62   */
63  public class DateUtil {
64  
65      /**
66       * Date format pattern used to parse HTTP date headers in RFC 1123 format.
67       */
68      public static final String PATTERN_RFC1123 = "EEE, dd MMM yyyy HH:mm:ss zzz";
69  
70      /**
71       * Date format pattern used to parse HTTP date headers in RFC 1036 format.
72       */
73      public static final String PATTERN_RFC1036 = "EEEE, dd-MMM-yy HH:mm:ss zzz";
74  
75      /**
76       * Date format pattern used to parse HTTP date headers in ANSI C
77       * <code>asctime()</code> format.
78       */
79      public static final String PATTERN_ASCTIME = "EEE MMM d HH:mm:ss yyyy";
80  
81      private static final Collection DEFAULT_PATTERNS = Arrays.asList(
82              new String[]{PATTERN_ASCTIME, PATTERN_RFC1036, PATTERN_RFC1123});
83  
84      private static final Date DEFAULT_TWO_DIGIT_YEAR_START;
85  
86      static {
87          Calendar calendar = Calendar.getInstance();
88          calendar.set(2000, Calendar.JANUARY, 1, 0, 0);
89          DEFAULT_TWO_DIGIT_YEAR_START = calendar.getTime();
90      }
91  
92      private static final TimeZone GMT = TimeZone.getTimeZone("GMT");
93  
94      /**
95       * Parses a date value.  The formats used for parsing the date value are retrieved from
96       * the default http params.
97       *
98       * @param dateValue the date value to parse
99       * @return the parsed date
100      * @throws DateParseException if the value could not be parsed using any of the
101      *                            supported date formats
102      */
103     public static Date parseDate(String dateValue) throws DateParseException {
104         return parseDate(dateValue, null, null);
105     }
106 
107     /**
108      * Parses the date value using the given date formats.
109      *
110      * @param dateValue   the date value to parse
111      * @param dateFormats the date formats to use
112      * @return the parsed date
113      * @throws DateParseException if none of the dataFormats could parse the dateValue
114      */
115     public static Date parseDate(String dateValue, Collection dateFormats)
116             throws DateParseException {
117         return parseDate(dateValue, dateFormats, null);
118     }
119 
120     /**
121      * Parses the date value using the given date formats.
122      *
123      * @param dateValue   the date value to parse
124      * @param dateFormats the date formats to use
125      * @param startDate   During parsing, two digit years will be placed in the range
126      *                    <code>startDate</code> to <code>startDate + 100 years</code>. This value may
127      *                    be <code>null</code>. When <code>null</code> is given as a parameter, year
128      *                    <code>2000</code> will be used.
129      * @return the parsed date
130      * @throws DateParseException if none of the dataFormats could parse the dateValue
131      */
132     public static Date parseDate(
133             String dateValue,
134             Collection dateFormats,
135             Date startDate
136     ) throws DateParseException {
137 
138         if (dateValue == null) {
139             throw new IllegalArgumentException("dateValue is null");
140         }
141         if (dateFormats == null) {
142             dateFormats = DEFAULT_PATTERNS;
143         }
144         if (startDate == null) {
145             startDate = DEFAULT_TWO_DIGIT_YEAR_START;
146         }
147         // trim single quotes around date if present
148         // see issue #5279
149         if (dateValue.length() > 1
150                 && dateValue.startsWith("'")
151                 && dateValue.endsWith("'")
152                 ) {
153             dateValue = dateValue.substring(1, dateValue.length() - 1);
154         }
155 
156         SimpleDateFormat dateParser = null;
157         Iterator formatIter = dateFormats.iterator();
158 
159         while (formatIter.hasNext()) {
160             String format = (String) formatIter.next();
161             if (dateParser == null) {
162                 dateParser = new SimpleDateFormat(format, Locale.US);
163                 dateParser.setTimeZone(TimeZone.getTimeZone("GMT"));
164                 dateParser.set2DigitYearStart(startDate);
165             } else {
166                 dateParser.applyPattern(format);
167             }
168             try {
169                 return dateParser.parse(dateValue);
170             } catch (ParseException pe) {
171                 // ignore this exception, we will try the next format
172             }
173         }
174 
175         // we were unable to parse the date
176         throw new DateParseException("Unable to parse the date " + dateValue);
177     }
178 
179     /**
180      * Formats the given date according to the RFC 1123 pattern.
181      *
182      * @param date The date to format.
183      * @return An RFC 1123 formatted date string.
184      * @see #PATTERN_RFC1123
185      */
186     public static String formatDate(Date date) {
187         return formatDate(date, PATTERN_RFC1123);
188     }
189 
190     /**
191      * Formats the given date according to the specified pattern.  The pattern
192      * must conform to that used by the {@link java.text.SimpleDateFormat simple date
193      * format} class.
194      *
195      * @param date    The date to format.
196      * @param pattern The pattern to use for formatting the date.
197      * @return A formatted date string.
198      * @throws IllegalArgumentException If the given date pattern is invalid.
199      * @see java.text.SimpleDateFormat
200      */
201     public static String formatDate(Date date, String pattern) {
202         if (date == null) throw new IllegalArgumentException("date is null");
203         if (pattern == null) throw new IllegalArgumentException("pattern is null");
204 
205         SimpleDateFormat formatter = new SimpleDateFormat(pattern, Locale.US);
206         formatter.setTimeZone(GMT);
207         return formatter.format(date);
208     }
