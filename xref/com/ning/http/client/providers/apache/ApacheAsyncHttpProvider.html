158         this.config = config;

169     private ScheduledExecutorService getReaper(AsyncHttpProviderConfig<?, ?> providerConfig) {
170 
171         ScheduledExecutorService reaper = null;
172         if (providerConfig instanceof ApacheAsyncHttpProvider) {
173             reaper = ApacheAsyncHttpProviderConfig.class.cast(providerConfig).getReaper();
174         }
175 
176         if (reaper == null)
177             reaper = Executors.newScheduledThreadPool(Runtime.getRuntime().availableProcessors(), new ThreadFactory() {
178                 public Thread newThread(Runnable r) {
179                     Thread t = new Thread(r, "AsyncHttpClient-Reaper");
180                     t.setDaemon(true);
181                     return t;
182                 }
183             });
184 
185         return reaper;
186     }

188     public <T> ListenableFuture<T> execute(Request request, AsyncHandler<T> handler) throws IOException {
189         if (isClose.get()) {
190             throw new IOException("Closed");
191         }
192 
193         if (handler instanceof ResumableAsyncHandler) {
194             request = ResumableAsyncHandler.class.cast(handler).adjustRequestRange(request);
195         }
196 
197         if (config.getMaxTotalConnections() > -1 && (maxConnections.get() + 1) > config.getMaxTotalConnections()) {
198             throw new IOException(String.format("Too many connections %s", config.getMaxTotalConnections()));
199         }
200 
201         if (idleConnectionTimeoutThread != null) {
202             idleConnectionTimeoutThread.shutdown();
203             idleConnectionTimeoutThread = null;
204         }
205 
206         int requestTimeout = requestTimeout(config, request.getPerRequestConfig());
207         if (config.getIdleConnectionTimeoutInMs() > 0 && requestTimeout != -1 && requestTimeout < config.getIdleConnectionTimeoutInMs()) {
208             idleConnectionTimeoutThread = new IdleConnectionTimeoutThread();
209             idleConnectionTimeoutThread.setConnectionTimeout(config.getIdleConnectionTimeoutInMs());
210             idleConnectionTimeoutThread.addConnectionManager(connectionManager);
211             idleConnectionTimeoutThread.start();
212         }
213 
214         HttpClient httpClient = new HttpClient(params, connectionManager);
215 
216         Realm realm = request.getRealm() != null ? request.getRealm() : config.getRealm();
217         if (realm != null) {
218             httpClient.getParams().setAuthenticationPreemptive(realm.getUsePreemptiveAuth());
219             Credentials defaultcreds = new UsernamePasswordCredentials(realm.getPrincipal(), realm.getPassword());
220             httpClient.getState().setCredentials(new AuthScope(null, -1, AuthScope.ANY_REALM), defaultcreds);
221         }
222 
223         HttpMethodBase method = createMethod(httpClient, request);
224         ApacheResponseFuture f = new ApacheResponseFuture<T>(handler, requestTimeout, request, method);
225         f.touch();
226 
227         f.setInnerFuture(config.executorService().submit(new ApacheClientRunnable(request, handler, method, f, httpClient)));
228         maxConnections.incrementAndGet();
229         return f;
230     }

232     public void close() {
233         reaper.shutdown();
234         if (idleConnectionTimeoutThread != null) {
235             idleConnectionTimeoutThread.shutdown();
236             idleConnectionTimeoutThread = null;
237         }
238         if (connectionManager != null) {
239             try {
240                 connectionManager.shutdown();
241             } catch (Exception e) {
242                 logger.error("Error shutting down connection manager", e);
243             }
244         }
245     }

251     private HttpMethodBase createMethod(HttpClient client, Request request) throws IOException, FileNotFoundException {
252         String methodName = request.getMethod();
253         HttpMethodBase method = null;
254         if (methodName.equalsIgnoreCase("POST") || methodName.equalsIgnoreCase("PUT")) {
255             EntityEnclosingMethod post = methodName.equalsIgnoreCase("POST") ? new PostMethod(request.getUrl()) : new PutMethod(request.getUrl());
256 
257             String bodyCharset = request.getBodyEncoding() == null ? DEFAULT_CHARSET : request.getBodyEncoding();
258 
259             post.getParams().setContentCharset("ISO-8859-1");
260             if (request.getByteData() != null) {
261                 post.setRequestEntity(new ByteArrayRequestEntity(request.getByteData()));
262                 post.setRequestHeader("Content-Length", String.valueOf(request.getByteData().length));
263             } else if (request.getStringData() != null) {
264                 post.setRequestEntity(new StringRequestEntity(request.getStringData(), "text/xml", bodyCharset));
265                 post.setRequestHeader("Content-Length", String.valueOf(request.getStringData().getBytes(bodyCharset).length));
266             } else if (request.getStreamData() != null) {
267                 InputStreamRequestEntity r = new InputStreamRequestEntity(request.getStreamData());
268                 post.setRequestEntity(r);
269                 post.setRequestHeader("Content-Length", String.valueOf(r.getContentLength()));
270 
271             } else if (request.getParams() != null) {
272                 StringBuilder sb = new StringBuilder();
273                 for (final Map.Entry<String, List<String>> paramEntry : request.getParams()) {
274                     final String key = paramEntry.getKey();
275                     for (final String value : paramEntry.getValue()) {
276                         if (sb.length() > 0) {
277                             sb.append("&");
278                         }
279                         UTF8UrlEncoder.appendEncoded(sb, key);
280                         sb.append("=");
281                         UTF8UrlEncoder.appendEncoded(sb, value);
282                     }
283                 }
284 
285                 post.setRequestHeader("Content-Length", String.valueOf(sb.length()));
286                 post.setRequestEntity(new StringRequestEntity(sb.toString(), "text/xml", "ISO-8859-1"));
287 
288                 if (!request.getHeaders().containsKey("Content-Type")) {
289                     post.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
290                 }
291             } else if (request.getParts() != null) {
292                 MultipartRequestEntity mre = createMultipartRequestEntity(bodyCharset, request.getParts(), post.getParams());
293                 post.setRequestEntity(mre);
294                 post.setRequestHeader("Content-Type", mre.getContentType());
295                 post.setRequestHeader("Content-Length", String.valueOf(mre.getContentLength()));
296             } else if (request.getEntityWriter() != null) {
297                 post.setRequestEntity(new EntityWriterRequestEntity(request.getEntityWriter(), computeAndSetContentLength(request, post)));
298             } else if (request.getFile() != null) {
299                 File file = request.getFile();
300                 if (!file.isFile()) {
301                     throw new IOException(String.format(Thread.currentThread()
302                             + "File %s is not a file or doesn't exist", file.getAbsolutePath()));
303                 }
304                 post.setRequestHeader("Content-Length", String.valueOf(file.length()));
305 
306                 FileInputStream fis = new FileInputStream(file);
307                 try {
308                     InputStreamRequestEntity r = new InputStreamRequestEntity(fis);
309                     post.setRequestEntity(r);
310                     post.setRequestHeader("Content-Length", String.valueOf(r.getContentLength()));
311                 } finally {
312                     fis.close();
313                 }
314             } else if (request.getBodyGenerator() != null) {
315                 Body body = request.getBodyGenerator().createBody();
316                 try {
317                     int length = (int) body.getContentLength();
318                     if (length < 0) {
319                         length = (int) request.getContentLength();
320                     }
321 
322                     // TODO: This is suboptimal
323                     if (length >= 0) {
324                         post.setRequestHeader("Content-Length", String.valueOf(length));
325 
326                         // This is totally sub optimal
327                         byte[] bytes = new byte[length];
328                         ByteBuffer buffer = ByteBuffer.wrap(bytes);
329                         for (; ; ) {
330                             buffer.clear();
331                             if (body.read(buffer) < 0) {
332                                 break;
333                             }
334                         }
335                         post.setRequestEntity(new ByteArrayRequestEntity(bytes));
336                     }
337                 } finally {
338                     try {
339                         body.close();
340                     } catch (IOException e) {
341                         logger.warn("Failed to close request body: {}", e.getMessage(), e);
342                     }
343                 }
344             }
345 
346             if (request.getHeaders().getFirstValue("Expect") != null
347                     && request.getHeaders().getFirstValue("Expect").equalsIgnoreCase("100-Continue")) {
348                 post.setUseExpectHeader(true);
349             }
350             method = post;
351         } else if (methodName.equalsIgnoreCase("DELETE")) {
352             method = new DeleteMethod(request.getUrl());
353         } else if (methodName.equalsIgnoreCase("HEAD")) {
354             method = new HeadMethod(request.getUrl());
355         } else if (methodName.equalsIgnoreCase("GET")) {
356             method = new GetMethod(request.getUrl());
357         } else if (methodName.equalsIgnoreCase("OPTIONS")) {
358             method = new OptionsMethod(request.getUrl());
359         } else {
360             throw new IllegalStateException(String.format("Invalid Method", methodName));
361         }
362 
363         ProxyServer proxyServer = ProxyUtils.getProxyServer(config, request);
364         if (proxyServer != null) {
365 
366             if (proxyServer.getPrincipal() != null) {
367                 Credentials defaultcreds = new UsernamePasswordCredentials(proxyServer.getPrincipal(), proxyServer.getPassword());
368                 client.getState().setProxyCredentials(new AuthScope(null, -1, AuthScope.ANY_REALM), defaultcreds);
369             }
370 
371             ProxyHost proxyHost = proxyServer == null ? null : new ProxyHost(proxyServer.getHost(), proxyServer.getPort());
372             client.getHostConfiguration().setProxyHost(proxyHost);
373         }
374         if(request.getLocalAddress()!=null) {
375             client.getHostConfiguration().setLocalAddress(request.getLocalAddress());
376         }
377 
378         method.setFollowRedirects(false);
379         if (isNonEmpty(request.getCookies())) {
380             method.setRequestHeader("Cookie", CookieEncoder.encode(request.getCookies()));
381         }
382 
383         if (request.getHeaders() != null) {
384             for (String name : request.getHeaders().keySet()) {
385                 if (!"host".equalsIgnoreCase(name)) {
386                     for (String value : request.getHeaders().get(name)) {
387                         method.setRequestHeader(name, value);
388                     }
389                 }
390             }
391         }
392 
393         if (request.getHeaders().getFirstValue("User-Agent") != null) {
394             method.setRequestHeader("User-Agent", request.getHeaders().getFirstValue("User-Agent"));
395         } else if (config.getUserAgent() != null) {
396             method.setRequestHeader("User-Agent", config.getUserAgent());
397         } else {
398             method.setRequestHeader("User-Agent", AsyncHttpProviderUtils.constructUserAgent(ApacheAsyncHttpProvider.class));
399         }
400 
401         if (config.isCompressionEnabled()) {
402             Header acceptableEncodingHeader = method.getRequestHeader("Accept-Encoding");
403             if (acceptableEncodingHeader != null) {
404                 String acceptableEncodings = acceptableEncodingHeader.getValue();
405                 if (acceptableEncodings.indexOf("gzip") == -1) {
406                     StringBuilder buf = new StringBuilder(acceptableEncodings);
407                     if (buf.length() > 1) {
408                         buf.append(",");
409                     }
410                     buf.append("gzip");
411                     method.setRequestHeader("Accept-Encoding", buf.toString());
412                 }
413             } else {
414                 method.setRequestHeader("Accept-Encoding", "gzip");
415             }
416         }
417 
418         if (request.getVirtualHost() != null) {
419 
420             String vs = request.getVirtualHost();
421             int index = vs.indexOf(":");
422             if (index > 0) {
423                 vs = vs.substring(0, index);
424             }
425             method.getParams().setVirtualHost(vs);
426         }
427 
428         return method;
429     }

431     private final static int computeAndSetContentLength(Request request, HttpMethodBase m) {
432         int lenght = (int) request.getContentLength();
433         if (lenght == -1 && m.getRequestHeader("Content-Length") != null) {
434             lenght = Integer.valueOf(m.getRequestHeader("Content-Length").getValue());
435         }
436 
437         if (lenght != -1) {
438             m.setRequestHeader("Content-Length", String.valueOf(lenght));
439         }
440         return lenght;
441     }

443     public class ApacheClientRunnable<T> implements Callable<T> {
444 
445         private final AsyncHandler<T> asyncHandler;
446         private HttpMethodBase method;
447         private final ApacheResponseFuture<T> future;
448         private Request request;
449         private final HttpClient httpClient;
450         private int currentRedirectCount;
451         private AtomicBoolean isAuth = new AtomicBoolean(false);
452         private boolean terminate = true;
453 
454         public ApacheClientRunnable(Request request, AsyncHandler<T> asyncHandler, HttpMethodBase method, ApacheResponseFuture<T> future, HttpClient httpClient) {
455             this.asyncHandler = asyncHandler;
456             this.method = method;
457             this.future = future;
458             this.request = request;
459             this.httpClient = httpClient;
460         }
461 
462         public T call() {
463             terminate = true;
464             AsyncHandler.STATE state = AsyncHandler.STATE.ABORT;
465             try {
466                 URI uri = null;
467                 try {
468                     uri = AsyncHttpProviderUtils.createUri(request.getRawUrl());
469                 } catch (IllegalArgumentException u) {
470                     uri = AsyncHttpProviderUtils.createUri(request.getUrl());
471                 }
472 
473                 int delay = requestTimeout(config, future.getRequest().getPerRequestConfig());
474                 if (delay != -1) {
475                     ReaperFuture reaperFuture = new ReaperFuture(future);
476                     Future scheduledFuture = reaper.scheduleAtFixedRate(reaperFuture, delay, 500, TimeUnit.MILLISECONDS);
477                     reaperFuture.setScheduledFuture(scheduledFuture);
478                     future.setReaperFuture(reaperFuture);
479                 }
480 
481                 if (asyncHandler instanceof TransferCompletionHandler) {
482                     throw new IllegalStateException(TransferCompletionHandler.class.getName() + "not supported by this provider");
483                 }
484 
485                 int statusCode = 200;
486                 try {
487                     statusCode = httpClient.executeMethod(method);
488                 } catch (CircularRedirectException ex) {
489                     // Quite ugly, but this is needed to unify 
490                     statusCode = 302;
491                     currentRedirectCount = config.getMaxRedirects();
492                 }
493 
494                 ApacheResponseStatus status = new ApacheResponseStatus(uri, method, ApacheAsyncHttpProvider.this);
495                 FilterContext fc = new FilterContext.FilterContextBuilder().asyncHandler(asyncHandler).request(request).responseStatus(status).build();
496                 for (ResponseFilter asyncFilter : config.getResponseFilters()) {
497                     fc = asyncFilter.filter(fc);
498                     if (fc == null) {
499                         throw new NullPointerException("FilterContext is null");
500                     }
501                 }
502 
503                 // The request has changed
504                 if (fc.replayRequest()) {
505                     request = fc.getRequest();
506                     method = createMethod(httpClient, request);
507                     terminate = false;
508                     return call();
509                 }
510 
511                 logger.debug("\n\nRequest {}\n\nResponse {}\n", request, method);
512 
513                 boolean redirectEnabled = (request.isRedirectEnabled() || config.isRedirectEnabled());
514                 if (redirectEnabled && (statusCode == 302 || statusCode == 301)) {
515 
516                     isAuth.set(false);
517 
518                     if (currentRedirectCount++ < config.getMaxRedirects()) {
519                         String location = method.getResponseHeader("Location").getValue();
520                         URI rediUri = AsyncHttpProviderUtils.getRedirectUri(uri, location);
521                         String newUrl = rediUri.toString();
522 
523                         if (!newUrl.equals(uri.toString())) {
524                             RequestBuilder builder = new RequestBuilder(request);
525 
526                             logger.debug("Redirecting to {}", newUrl);
527 
528                             request = builder.setUrl(newUrl).build();
529                             method = createMethod(httpClient, request);
530                             terminate = false;
531                             return call();
532                         }
533                     } else {
534                         throw new MaxRedirectException("Maximum redirect reached: " + config.getMaxRedirects());
535                     }
536                 }
537 
538                 state = asyncHandler.onStatusReceived(status);
539                 if (state == AsyncHandler.STATE.CONTINUE) {
540                     state = asyncHandler.onHeadersReceived(new ApacheResponseHeaders(uri, method, ApacheAsyncHttpProvider.this));
541                 }
542 
543                 if (state == AsyncHandler.STATE.CONTINUE) {
544                     InputStream is = method.getResponseBodyAsStream();
545                     if (is != null) {
546                         Header h = method.getResponseHeader("Content-Encoding");
547                         if (h != null) {
548                             String contentEncoding = h.getValue();
549                             boolean isGZipped = contentEncoding == null ? false : "gzip".equalsIgnoreCase(contentEncoding);
550                             if (isGZipped) {
551                                 is = new GZIPInputStream(is);
552                             }
553                         }
554 
555                         int byteToRead = (int) method.getResponseContentLength();
556                         InputStream stream = is;
557                         if (byteToRead <= 0) {
558                             int[] lengthWrapper = new int[1];
559                             byte[] bytes = AsyncHttpProviderUtils.readFully(is, lengthWrapper);
560                             stream = new ByteArrayInputStream(bytes, 0, lengthWrapper[0]);
561                             byteToRead = lengthWrapper[0];
562                         }
563 
564                         if (byteToRead > 0) {
565                             int minBytes = Math.min(8192, byteToRead);
566                             byte[] bytes = new byte[minBytes];
567                             int leftBytes = minBytes < 8192 ? minBytes : byteToRead;
568                             int read = 0;
569                             while (leftBytes > -1) {
570 
571                                 try {
572                                     read = stream.read(bytes);
573                                 } catch (IOException ex) {
574                                     logger.warn("Connection closed", ex);
575                                     read = -1;
576                                 }
577 
578                                 if (read == -1) {
579                                     break;
580                                 }
581 
582                                 future.touch();
583 
584                                 byte[] b = new byte[read];
585                                 System.arraycopy(bytes, 0, b, 0, read);
586                                 leftBytes -= read;
587 
588                                 asyncHandler.onBodyPartReceived(new ApacheResponseBodyPart(uri, b, ApacheAsyncHttpProvider.this, leftBytes > -1));
589 
590                             }
591                         }
592                     }
593 
594                     if (method.getName().equalsIgnoreCase("HEAD")) {
595                         asyncHandler.onBodyPartReceived(new ApacheResponseBodyPart(uri, "".getBytes(), ApacheAsyncHttpProvider.this, true));
596                     }
597                 }
598 
599                 if (asyncHandler instanceof ProgressAsyncHandler) {
600                 	ProgressAsyncHandler progressAsyncHandler = (ProgressAsyncHandler) asyncHandler;
601                 	progressAsyncHandler.onHeaderWriteCompleted();
602                 	progressAsyncHandler.onContentWriteCompleted();
603                 }
604 
605                 try {
606                     return asyncHandler.onCompleted();
607                 } catch (Throwable t) {
608                     RuntimeException ex = new RuntimeException();
609                     ex.initCause(t);
610                     throw ex;
611                 }
612             } catch (Throwable t) {
613 
614                 if (t instanceof IOException && !config.getIOExceptionFilters().isEmpty()) {
615                     FilterContext fc = new FilterContext.FilterContextBuilder().asyncHandler(asyncHandler)
616                             .request(future.getRequest()).ioException(IOException.class.cast(t)).build();
617 
618                     try {
619                         fc = handleIoException(fc);
620                     } catch (FilterException e) {
621                         if (config.getMaxTotalConnections() != -1) {
622                             maxConnections.decrementAndGet();
623                         }
624                         future.done();
625                         method.releaseConnection();
626                     }
627 
628                     if (fc.replayRequest()) {
629                         request = fc.getRequest();
630                         return call();
631                     }
632                 }
633 
634                 if (method.isAborted()) {
635                     return null;
636                 }
637 
638                 logger.debug(t.getMessage(), t);
639 
640                 try {
641                     future.abort(filterException(t));
642                 } catch (Throwable t2) {
643                     logger.error(t2.getMessage(), t2);
644                 }
645             } finally {
646                 if (terminate) {
647                     if (config.getMaxTotalConnections() != -1) {
648                         maxConnections.decrementAndGet();
649                     }
650                     future.done();
651 
652                     // Crappy Apache HttpClient who blocks forever here with large files.
653                     config.executorService().submit(new Runnable() {
654 
655                         public void run() {
656                             method.releaseConnection();
657                         }
658                     });
659                 }
660             }
661             return null;
662         }
663 
664         private Throwable filterException(Throwable t) {
665             if (t instanceof UnknownHostException) {
666                 t = new ConnectException(t.getMessage());
667 
668             } else if (t instanceof NoHttpResponseException) {
669                 int responseTimeoutInMs = config.getRequestTimeoutInMs();
670 
671                 if (request.getPerRequestConfig() != null && request.getPerRequestConfig().getRequestTimeoutInMs() != -1) {
672                     responseTimeoutInMs = request.getPerRequestConfig().getRequestTimeoutInMs();
673                 }
674                 t = new TimeoutException(String.format("No response received after %s", responseTimeoutInMs));
675 
676             } else if (t instanceof SSLHandshakeException) {
677                 Throwable t2 = new ConnectException();
678                 t2.initCause(t);
679                 t = t2;
680             }
681 
682             return t;
683         }

686             for (IOExceptionFilter asyncFilter : config.getIOExceptionFilters()) {
687                 fc = asyncFilter.filter(fc);
688                 if (fc == null) {
689                     throw new NullPointerException("FilterContext is null");
690                 }
691             }
692             return fc;
693         }

696     private MultipartRequestEntity createMultipartRequestEntity(String charset, List<Part> params, HttpMethodParams methodParams) throws FileNotFoundException {
697         org.apache.commons.httpclient.methods.multipart.Part[] parts = new org.apache.commons.httpclient.methods.multipart.Part[params.size()];
698         int i = 0;
699 
700         for (Part part : params) {
701             if (part instanceof StringPart) {
702                 parts[i] = new org.apache.commons.httpclient.methods.multipart.StringPart(part.getName(),
703                         ((StringPart) part).getValue(),
704                         charset);
705             } else if (part instanceof FilePart) {
706                 parts[i] = new org.apache.commons.httpclient.methods.multipart.FilePart(part.getName(),
707                         ((FilePart) part).getFile(),
708                         ((FilePart) part).getMimeType(),
709                         ((FilePart) part).getCharSet());
710 
711             } else if (part instanceof ByteArrayPart) {
712                 PartSource source = new ByteArrayPartSource(((ByteArrayPart) part).getFileName(), ((ByteArrayPart) part).getData());
713                 parts[i] = new org.apache.commons.httpclient.methods.multipart.FilePart(part.getName(),
714                         source,

716                         ((ByteArrayPart) part).getCharSet());
717 
718             } else if (part == null) {
719                 throw new NullPointerException("Part cannot be null");
720             } else {
721                 throw new IllegalArgumentException(String.format("Unsupported part type for multipart parameter %s",
722                         part.getName()));
723             }
724             ++i;
725         }
