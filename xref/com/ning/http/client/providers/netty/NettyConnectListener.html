44  /**
45   * Non Blocking connect.
46   */
47  final class NettyConnectListener<T> implements ChannelFutureListener {
48      private final static Logger logger = LoggerFactory.getLogger(NettyConnectListener.class);
49      private final AsyncHttpClientConfig config;
50      private final NettyResponseFuture<T> future;
51      private final HttpRequest nettyRequest;
52      private final AtomicBoolean handshakeDone = new AtomicBoolean(false);
53  
54      private NettyConnectListener(AsyncHttpClientConfig config,
55                                   NettyResponseFuture<T> future) {
56          this.config = config;
57          this.future = future;
58          this.nettyRequest = future.getNettyRequest();
59      }
60  
61      public NettyResponseFuture<T> future() {
62          return future;
63      }
64  
65      public final void operationComplete(ChannelFuture f) throws Exception {
66          if (f.isSuccess()) {
67              Channel channel = f.getChannel();
68              channel.getPipeline().getContext(NettyAsyncHttpProvider.class).setAttachment(future);
69              SslHandler sslHandler = (SslHandler) channel.getPipeline().get(NettyAsyncHttpProvider.SSL_HANDLER);
70              if (!handshakeDone.getAndSet(true) && (sslHandler != null)) {
71                  ((SslHandler) channel.getPipeline().get(NettyAsyncHttpProvider.SSL_HANDLER)).handshake().addListener(this);
72                  return;
73              }
74  
75              HostnameVerifier v = config.getHostnameVerifier();
76              if (sslHandler != null && !(v instanceof AllowAllHostnameVerifier)) {
77                  // TODO: channel.getRemoteAddress()).getHostName() is very expensive. Should cache the result.
78                  if (!v.verify(InetSocketAddress.class.cast(channel.getRemoteAddress()).getHostName(),
79                          sslHandler.getEngine().getSession())) {
80                      throw new ConnectException("HostnameVerifier exception.");
81                  }
82              }
83  
84              future.provider().writeRequest(f.getChannel(), config, future);
85          } else {
86              Throwable cause = f.getCause();
87  
88              boolean canRetry = future.canRetry();
89              logger.debug("Trying to recover a dead cached channel {} with a retry value of {} ", f.getChannel(), canRetry);
90              if (canRetry && cause != null && (NettyAsyncHttpProvider.abortOnDisconnectException(cause)
91                      || cause instanceof ClosedChannelException
92                      || future.getState() != NettyResponseFuture.STATE.NEW)) {
93  
94                  logger.debug("Retrying {} ", nettyRequest);
95                  if (future.provider().remotelyClosed(f.getChannel(), future)) {
96                      return;
97                  }
98              }
99  
100             logger.debug("Failed to recover from exception: {} with channel {}", cause, f.getChannel());
101 
102             boolean printCause = f.getCause() != null && cause.getMessage() != null;
103             ConnectException e = new ConnectException(printCause ? cause.getMessage() + " to " + future.getURI().toString() : future.getURI().toString());
104             if (cause != null) {
105                 e.initCause(cause);
106             }
107             future.abort(e);
108         }
109     }

111     public static class Builder<T> {
112         private final AsyncHttpClientConfig config;
113 
114         private final Request request;
115         private final AsyncHandler<T> asyncHandler;
116         private NettyResponseFuture<T> future;
117         private final NettyAsyncHttpProvider provider;
118         private final ChannelBuffer buffer;
119 
120         public Builder(AsyncHttpClientConfig config, Request request, AsyncHandler<T> asyncHandler,
121                        NettyAsyncHttpProvider provider, ChannelBuffer buffer) {
122 
123             this.config = config;
124             this.request = request;
125             this.asyncHandler = asyncHandler;
126             this.future = null;
127             this.provider = provider;
128             this.buffer = buffer;
129         }
130 
131         public Builder(AsyncHttpClientConfig config, Request request, AsyncHandler<T> asyncHandler,
132                        NettyResponseFuture<T> future, NettyAsyncHttpProvider provider, ChannelBuffer buffer) {
133 
134             this.config = config;
135             this.request = request;
136             this.asyncHandler = asyncHandler;
137             this.future = future;
138             this.provider = provider;
139             this.buffer = buffer;
140         }
141 
142         public NettyConnectListener<T> build(final URI uri) throws IOException {
143             ProxyServer proxyServer = ProxyUtils.getProxyServer(config, request);
144             HttpRequest nettyRequest = NettyAsyncHttpProvider.buildRequest(config, request, uri, true, buffer, proxyServer);
145             if (future == null) {
146                 future = NettyAsyncHttpProvider.newFuture(uri, request, asyncHandler, nettyRequest, config, provider, proxyServer);
147             } else {
148                 future.setNettyRequest(nettyRequest);
149                 future.setRequest(request);
150             }
151             return new NettyConnectListener<T>(config, future);
152         }
153     }
