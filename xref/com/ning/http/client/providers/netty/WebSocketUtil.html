13  package com.ning.http.client.providers.netty;
14  
15  import com.ning.http.util.Base64;
16  
17  import java.io.UnsupportedEncodingException;
18  import java.security.MessageDigest;
19  import java.security.NoSuchAlgorithmException;
20  
21  public final class WebSocketUtil {
22      public static final String MAGIC_GUID = "258EAFA5-E914-47DA-95CA-C5AB0DC85B11";
23  
24      public static String getKey() {
25          byte[] nonce = createRandomBytes(16);
26          return base64Encode(nonce);
27      }
28  
29      public static String getAcceptKey(String key) throws UnsupportedEncodingException {
30          String acceptSeed = key + MAGIC_GUID;
31          byte[] sha1 = sha1(acceptSeed.getBytes("US-ASCII"));
32          return base64Encode(sha1);
33      }
34  
35      public static byte[] md5(byte[] bytes) {
36          try {
37              MessageDigest md = MessageDigest.getInstance("MD5");
38              return md.digest(bytes);
39          } catch (NoSuchAlgorithmException e) {
40              throw new InternalError("MD5 not supported on this platform");
41          }
42      }
43  
44      public static byte[] sha1(byte[] bytes) {
45          try {
46              MessageDigest md = MessageDigest.getInstance("SHA1");
47              return md.digest(bytes);
48          } catch (NoSuchAlgorithmException e) {
49              throw new InternalError("SHA-1 not supported on this platform");
50          }
51      }
52  
53      public static String base64Encode(byte[] bytes) {
54          return Base64.encode(bytes);
55      }
56  
57      public static byte[] createRandomBytes(int size) {
58          byte[] bytes = new byte[size];
59  
60          for (int i = 0; i < size; i++) {
61              bytes[i] = (byte) createRandomNumber(0, 255);
62          }
63  
64          return bytes;
65      }
66  
67      public static int createRandomNumber(int min, int max) {
68          return (int) (Math.random() * max + min);
69      }
70  
71  }
