127     public <T> ListenableFuture<T> execute(Request request, AsyncHandler<T> handler, ListenableFuture<?> future) throws IOException {
128         if (isClose.get()) {
129             throw new IOException("Closed");
130         }
131 
132         if (config.getMaxTotalConnections() > -1 && (maxConnections.get() + 1) > config.getMaxTotalConnections()) {
133             throw new IOException(String.format("Too many connections %s", config.getMaxTotalConnections()));
134         }
135 
136         ProxyServer proxyServer = ProxyUtils.getProxyServer(config, request);
137         Realm realm = request.getRealm() != null ? request.getRealm() : config.getRealm();
138         Proxy proxy = null;
139         if (proxyServer != null || realm != null) {
140             try {
141                 proxy = configureProxyAndAuth(proxyServer, realm);
142             } catch (AuthenticationException e) {
143                 throw new IOException(e.getMessage());
144             }
145         }
146 
147         HttpURLConnection urlConnection = createUrlConnection(request);
148 
149         PerRequestConfig conf = request.getPerRequestConfig();
150         int requestTimeout = (conf != null && conf.getRequestTimeoutInMs() != 0) ?
151                 conf.getRequestTimeoutInMs() : config.getRequestTimeoutInMs();
152 
153         JDKDelegateFuture delegate = null;
154         if (future != null) {
155             delegate = new JDKDelegateFuture(handler, requestTimeout, future, urlConnection);
156         }
157 
158         JDKFuture f = delegate == null ? new JDKFuture<T>(handler, requestTimeout, urlConnection) : delegate;
159         f.touch();
160 
161         f.setInnerFuture(config.executorService().submit(new AsyncHttpUrlConnection(urlConnection, request, handler, f)));
162         maxConnections.incrementAndGet();
163 
164         return f;
165     }

167     private HttpURLConnection createUrlConnection(Request request) throws IOException {
168         ProxyServer proxyServer = ProxyUtils.getProxyServer(config, request);
169         Realm realm = request.getRealm() != null ? request.getRealm() : config.getRealm();
170         Proxy proxy = null;
171         if (proxyServer != null || realm != null) {
172             try {
173                 proxy = configureProxyAndAuth(proxyServer, realm);
174             } catch (AuthenticationException e) {
175                 throw new IOException(e.getMessage());
176             }
177         }
178 
179         HttpURLConnection urlConnection = (HttpURLConnection)
180             request.getURI().toURL().openConnection(proxy == null ? Proxy.NO_PROXY : proxy);
181 
182         if (request.getUrl().startsWith("https")) {
183             HttpsURLConnection secure = (HttpsURLConnection) urlConnection;
184             SSLContext sslContext = config.getSSLContext();
185             if (sslContext == null) {
186                 try {
187                     sslContext = SslUtils.getSSLContext();
188                 } catch (NoSuchAlgorithmException e) {
189                     throw new IOException(e.getMessage());
190                 } catch (GeneralSecurityException e) {
191                     throw new IOException(e.getMessage());
192                 }
193             }
194             secure.setSSLSocketFactory(sslContext.getSocketFactory());
195             secure.setHostnameVerifier(config.getHostnameVerifier());
196         }
197         return urlConnection;
198     }

208     private final class AsyncHttpUrlConnection<T> implements Callable<T> {
209 
210         private HttpURLConnection urlConnection;
211         private Request request;
212         private final AsyncHandler<T> asyncHandler;
213         private final ListenableFuture<T> future;
214         private int currentRedirectCount;
215         private AtomicBoolean isAuth = new AtomicBoolean(false);
216         private byte[] cachedBytes;
217         private int cachedBytesLenght;
218         private boolean terminate = true;
219 
220         public AsyncHttpUrlConnection(HttpURLConnection urlConnection, Request request, AsyncHandler<T> asyncHandler, ListenableFuture<T> future) {
221             this.urlConnection = urlConnection;
222             this.request = request;
223             this.asyncHandler = asyncHandler;
224             this.future = future;
225             this.request = request;
226         }
227 
228         public T call() throws Exception {
229             AsyncHandler.STATE state = AsyncHandler.STATE.ABORT;
230             try {
231                 URI uri = null;
232                 // Encoding with URLConnection is a bit bogus so we need to try both way before setting it
233                 try {
234                     uri = AsyncHttpProviderUtils.createUri(request.getRawUrl());
235                 } catch (IllegalArgumentException u) {
236                     uri = AsyncHttpProviderUtils.createUri(request.getUrl());
237                 }
238 
239                 configure(uri, urlConnection, request);
240                 urlConnection.connect();
241 
242                 if (asyncHandler instanceof TransferCompletionHandler) {
243                     throw new IllegalStateException(TransferCompletionHandler.class.getName() + "not supported by this provider");
244                 }
245 
246                 int statusCode = urlConnection.getResponseCode();
247 
248                 logger.debug("\n\nRequest {}\n\nResponse {}\n", request, statusCode);
249 
250                 ResponseStatus status = new ResponseStatus(uri, urlConnection, JDKAsyncHttpProvider.this);
251                 FilterContext fc = new FilterContext.FilterContextBuilder().asyncHandler(asyncHandler).request(request).responseStatus(status).build();
252                 for (ResponseFilter asyncFilter : config.getResponseFilters()) {
253                     fc = asyncFilter.filter(fc);
254                     if (fc == null) {
255                         throw new NullPointerException("FilterContext is null");
256                     }
257                 }
258 
259                 // The request has changed
260                 if (fc.replayRequest()) {
261                     request = fc.getRequest();
262                     urlConnection = createUrlConnection(request);
263                     terminate = false;
264                     return call();
265                 }
266 
267                 boolean redirectEnabled = (request.isRedirectEnabled() || config.isRedirectEnabled());
268                 if (redirectEnabled && (statusCode == 302 || statusCode == 301)) {
269 
270                     if (currentRedirectCount++ < config.getMaxRedirects()) {
271                         String location = urlConnection.getHeaderField("Location");
272                         URI redirUri = AsyncHttpProviderUtils.getRedirectUri(uri, location);
273                         String newUrl = redirUri.toString();
274 
275                         if (!newUrl.equals(uri.toString())) {
276                             RequestBuilder builder = new RequestBuilder(request);
277 
278                             logger.debug("Redirecting to {}", newUrl);
279 
280                             request = builder.setUrl(newUrl).build();
281                             urlConnection = createUrlConnection(request);
282                             terminate = false;
283                             return call();
284                         }
285                     } else {
286                         throw new MaxRedirectException("Maximum redirect reached: " + config.getMaxRedirects());
287                     }
288                 }
289 
290                 Realm realm = request.getRealm() != null ? request.getRealm() : config.getRealm();
291                 if (statusCode == 401 && !isAuth.getAndSet(true) && realm != null) {
292                     String wwwAuth = urlConnection.getHeaderField("WWW-Authenticate");
293 
294                     logger.debug("Sending authentication to {}", request.getUrl());
295 
296                     Realm nr = new Realm.RealmBuilder().clone(realm)
297                             .parseWWWAuthenticateHeader(wwwAuth)
298                             .setUri(URI.create(request.getUrl()).getPath())
299                             .setMethodName(request.getMethod())
300                             .setUsePreemptiveAuth(true)
301                             .build();
302                     RequestBuilder builder = new RequestBuilder(request);
303                     request = builder.setRealm(nr).build();
304                     urlConnection = createUrlConnection(request);
305                     terminate = false;
306                     return call();
307                 }
308 
309                 state = asyncHandler.onStatusReceived(status);
310                 if (state == AsyncHandler.STATE.CONTINUE) {
311                     state = asyncHandler.onHeadersReceived(new ResponseHeaders(uri, urlConnection, JDKAsyncHttpProvider.this));
312                 }
313 
314                 if (state == AsyncHandler.STATE.CONTINUE) {
315                     InputStream is = getInputStream(urlConnection);
316                     String contentEncoding = urlConnection.getHeaderField("Content-Encoding");
317                     boolean isGZipped = contentEncoding == null ? false : "gzip".equalsIgnoreCase(contentEncoding);
318                     if (isGZipped) {
319                         is = new GZIPInputStream(is);
320                     }
321 
322                     int byteToRead = urlConnection.getContentLength();
323                     InputStream stream = is;
324                     if (bufferResponseInMemory || byteToRead <= 0) {
325                         int[] lengthWrapper = new int[1];
326                         byte[] bytes = AsyncHttpProviderUtils.readFully(is, lengthWrapper);
327                         stream = new ByteArrayInputStream(bytes, 0, lengthWrapper[0]);
328                         byteToRead = lengthWrapper[0];
329                     }
330 
331                     if (byteToRead > 0) {
332                         int minBytes = Math.min(8192, byteToRead);
333                         byte[] bytes = new byte[minBytes];
334                         int leftBytes = minBytes < 8192 ? minBytes : byteToRead;
335                         int read = 0;
336                         while (leftBytes > -1) {
337 
338                             read = stream.read(bytes);
339                             if (read == -1) {
340                                 break;
341                             }
342 
343                             future.touch();
344 
345                             byte[] b = new byte[read];
346                             System.arraycopy(bytes, 0, b, 0, read);
347                             leftBytes -= read;
348                             asyncHandler.onBodyPartReceived(new ResponseBodyPart(uri, b, JDKAsyncHttpProvider.this, leftBytes > -1));
349                         }
350                     }
351 
352                     if (request.getMethod().equalsIgnoreCase("HEAD")) {
353                         asyncHandler.onBodyPartReceived(new ResponseBodyPart(uri, "".getBytes(), JDKAsyncHttpProvider.this, true));
354                     }
355                 }
356 
357                 if (asyncHandler instanceof ProgressAsyncHandler) {
358                 	ProgressAsyncHandler progressAsyncHandler = (ProgressAsyncHandler) asyncHandler;
359                 	progressAsyncHandler.onHeaderWriteCompleted();
360                 	progressAsyncHandler.onContentWriteCompleted();
361                 }
362                 try {
363                     T t = asyncHandler.onCompleted();
364                     future.content(t);
365                     future.done();
366                     return t;
367                 } catch (Throwable t) {
368                     RuntimeException ex = new RuntimeException();
369                     ex.initCause(t);
370                     throw ex;
371                 }
372             } catch (Throwable t) {
373                 logger.debug(t.getMessage(), t);
374 
375                 if (t instanceof IOException && !config.getIOExceptionFilters().isEmpty()) {
376                     FilterContext fc = new FilterContext.FilterContextBuilder().asyncHandler(asyncHandler)
377                             .request(request).ioException(IOException.class.cast(t)).build();
378 
379                     try {
380                         fc = handleIoException(fc);
381                     } catch (FilterException e) {
382                         if (config.getMaxTotalConnections() != -1) {
383                             maxConnections.decrementAndGet();
384                         }
385                         future.done();
386                     }
387 
388                     if (fc.replayRequest()) {
389                         request = fc.getRequest();
390                         urlConnection = createUrlConnection(request);
391                         return call();
392                     }
393                 }
394 
395                 try {
396                     future.abort(filterException(t));
397                 } catch (Throwable t2) {
398                     logger.error(t2.getMessage(), t2);
399                 }
400             } finally {
401                 if (terminate) {
402                     if (config.getMaxTotalConnections() != -1) {
403                         maxConnections.decrementAndGet();
404                     }
405                     urlConnection.disconnect();
406                     if (jdkNtlmDomain != null) {
407                         System.setProperty(NTLM_DOMAIN, jdkNtlmDomain);
408                     }
409                     Authenticator.setDefault(jdkAuthenticator);
410                 }
411             }
412             return null;
413         }
414 
415         private FilterContext handleIoException(FilterContext fc) throws FilterException {
416             for (IOExceptionFilter asyncFilter : config.getIOExceptionFilters()) {
417                 fc = asyncFilter.filter(fc);
418                 if (fc == null) {
419                     throw new NullPointerException("FilterContext is null");
420                 }
421             }
422             return fc;
423         }

425         private Throwable filterException(Throwable t) {
426             if (t instanceof UnknownHostException) {
427                 t = new ConnectException(t.getMessage());
428 
429             } else if (t instanceof SocketTimeoutException) {
430                 int responseTimeoutInMs = config.getRequestTimeoutInMs();
431 
432                 if (request.getPerRequestConfig() != null && request.getPerRequestConfig().getRequestTimeoutInMs() != -1) {
433                     responseTimeoutInMs = request.getPerRequestConfig().getRequestTimeoutInMs();
434                 }
435                 t = new TimeoutException(String.format("No response received after %s", responseTimeoutInMs));
436 
437             } else if (t instanceof SSLHandshakeException) {
438                 Throwable t2 = new ConnectException();
439                 t2.initCause(t);
440                 t = t2;
441             }
442 
443             return t;
444         }

446         private void configure(URI uri, HttpURLConnection urlConnection, Request request) throws IOException, AuthenticationException {
447 
448             PerRequestConfig conf = request.getPerRequestConfig();
449             int requestTimeout = (conf != null && conf.getRequestTimeoutInMs() != 0) ?
450                     conf.getRequestTimeoutInMs() : config.getRequestTimeoutInMs();
451 
452             urlConnection.setConnectTimeout(config.getConnectionTimeoutInMs());
453 
454             if (requestTimeout != -1)
455                 urlConnection.setReadTimeout(requestTimeout);
456 
457             urlConnection.setInstanceFollowRedirects(false);
458             String host = uri.getHost();
459             String method = request.getMethod();
460 
461             if (request.getVirtualHost() != null) {
462                 host = request.getVirtualHost();
463             }
464 
465             if (uri.getPort() == -1 || request.getVirtualHost() != null) {
466                 urlConnection.setRequestProperty("Host", host);
467             } else {
468                 urlConnection.setRequestProperty("Host", host + ":" + uri.getPort());
469             }
470 
471 
472             if (config.isCompressionEnabled()) {
473                 urlConnection.setRequestProperty("Accept-Encoding", "gzip");
474             }
475 
476             if (!method.equalsIgnoreCase("CONNECT")) {
477                 FluentCaseInsensitiveStringsMap h = request.getHeaders();
478                 if (h != null) {
479                     for (String name : h.keySet()) {
480                         if (!"host".equalsIgnoreCase(name)) {
481                             for (String value : h.get(name)) {
482                                 urlConnection.setRequestProperty(name, value);
483                                 if (name.equalsIgnoreCase("Expect")) {
484                                     throw new IllegalStateException("Expect: 100-Continue not supported");
485                                 }
486                             }
487                         }
488                     }
489                 }
490             }
491 
492             String ka = AsyncHttpProviderUtils.keepAliveHeaderValue(config);
493             urlConnection.setRequestProperty("Connection", ka);
494             ProxyServer proxyServer = ProxyUtils.getProxyServer(config, request);
495             boolean avoidProxy = ProxyUtils.avoidProxy(proxyServer, uri.getHost());
496             if (!avoidProxy) {
497                 urlConnection.setRequestProperty("Proxy-Connection", ka);
498                 if (proxyServer.getPrincipal() != null) {
499                     urlConnection.setRequestProperty("Proxy-Authorization", AuthenticatorUtils.computeBasicAuthentication(proxyServer));
500                 }
501 
502                 if (proxyServer.getProtocol().equals(ProxyServer.Protocol.NTLM)) {
503                     jdkNtlmDomain = System.getProperty(NTLM_DOMAIN);
504                     System.setProperty(NTLM_DOMAIN, proxyServer.getNtlmDomain());
505                 }
506             }
507 
508             Realm realm = request.getRealm() != null ? request.getRealm() : config.getRealm();
509             if (realm != null && realm.getUsePreemptiveAuth()) {
510                 switch (realm.getAuthScheme()) {
511                     case BASIC:
512                         urlConnection.setRequestProperty("Authorization",
513                                 AuthenticatorUtils.computeBasicAuthentication(realm));
514                         break;
515                     case DIGEST:
516                         if (isNonEmpty(realm.getNonce())) {
517                             try {
518                                 urlConnection.setRequestProperty("Authorization",
519                                         AuthenticatorUtils.computeDigestAuthentication(realm));
520                             } catch (NoSuchAlgorithmException e) {
521                                 throw new SecurityException(e);
522                             }
523                         }
524                         break;
525                     case NTLM:
526                         jdkNtlmDomain = System.getProperty(NTLM_DOMAIN);
527                         System.setProperty(NTLM_DOMAIN, realm.getDomain());
528                         break;
529                     case NONE:
530                         break;
531                     default:
532                         throw new IllegalStateException(String.format("Invalid Authentication %s", realm.toString()));
533                 }
534 
535             }
536 
537             // Add default accept headers.
538             if (request.getHeaders().getFirstValue("Accept") == null) {
539                 urlConnection.setRequestProperty("Accept", "*/*");
540             }
541 
542             if (request.getHeaders().getFirstValue("User-Agent") != null) {
543                 urlConnection.setRequestProperty("User-Agent", request.getHeaders().getFirstValue("User-Agent"));
544             } else if (config.getUserAgent() != null) {
545                 urlConnection.setRequestProperty("User-Agent", config.getUserAgent());
546             } else {
547                 urlConnection.setRequestProperty("User-Agent", AsyncHttpProviderUtils.constructUserAgent(JDKAsyncHttpProvider.class));
548             }
549 
550             if (isNonEmpty(request.getCookies())) {
551                 urlConnection.setRequestProperty("Cookie", CookieEncoder.encode(request.getCookies()));
552             }
553 
554             String reqType = request.getMethod();
555             urlConnection.setRequestMethod(reqType);
556 
557             if ("POST".equals(reqType) || "PUT".equals(reqType)) {
558                 urlConnection.setRequestProperty("Content-Length", "0");
559                 urlConnection.setDoOutput(true);
560                 String bodyCharset = request.getBodyEncoding() == null ? DEFAULT_CHARSET : request.getBodyEncoding();
561 
562                 if (cachedBytes != null) {
563                     urlConnection.setRequestProperty("Content-Length", String.valueOf(cachedBytesLenght));
564                     urlConnection.setFixedLengthStreamingMode(cachedBytesLenght);
565                     urlConnection.getOutputStream().write(cachedBytes, 0, cachedBytesLenght);
566                 } else if (request.getByteData() != null) {
567                     urlConnection.setRequestProperty("Content-Length", String.valueOf(request.getByteData().length));
568                     urlConnection.setFixedLengthStreamingMode(request.getByteData().length);
569 
570                     urlConnection.getOutputStream().write(request.getByteData());
571                 } else if (request.getStringData() != null) {
572                     if (!request.getHeaders().containsKey("Content-Type")) {
573                         urlConnection.setRequestProperty("Content-Type", "text/html;" + bodyCharset);
574                     }
575                     byte[] b = request.getStringData().getBytes(bodyCharset);
576                     urlConnection.setRequestProperty("Content-Length", String.valueOf(b.length));
577                     urlConnection.getOutputStream().write(b);
578                 } else if (request.getStreamData() != null) {
579                     int[] lengthWrapper = new int[1];
580                     cachedBytes = AsyncHttpProviderUtils.readFully(request.getStreamData(), lengthWrapper);
581                     cachedBytesLenght = lengthWrapper[0];
582                     urlConnection.setRequestProperty("Content-Length", String.valueOf(cachedBytesLenght));
583                     urlConnection.setFixedLengthStreamingMode(cachedBytesLenght);
584 
585                     urlConnection.getOutputStream().write(cachedBytes, 0, cachedBytesLenght);
586                 } else if (request.getParams() != null) {
587                     StringBuilder sb = new StringBuilder();
588                     for (final Map.Entry<String, List<String>> paramEntry : request.getParams()) {
589                         final String key = paramEntry.getKey();
590                         for (final String value : paramEntry.getValue()) {
591                             if (sb.length() > 0) {
592                                 sb.append("&");
593                             }
594                             UTF8UrlEncoder.appendEncoded(sb, key);
595                             sb.append("=");
596                             UTF8UrlEncoder.appendEncoded(sb, value);
597                         }
598                     }
599                     urlConnection.setRequestProperty("Content-Length", String.valueOf(sb.length()));
600                     urlConnection.setFixedLengthStreamingMode(sb.length());
601 
602                     if (!request.getHeaders().containsKey("Content-Type")) {
603                         urlConnection.setRequestProperty("Content-Type", "application/x-www-form-urlencoded");
604                     }
605                     urlConnection.getOutputStream().write(sb.toString().getBytes(bodyCharset));
606                 } else if (request.getParts() != null) {
607                     int lenght = (int) request.getContentLength();
608                     if (lenght != -1) {
609                         urlConnection.setRequestProperty("Content-Length", String.valueOf(lenght));
610                         urlConnection.setFixedLengthStreamingMode(lenght);
611                     }
612 
613                     if (lenght == -1) {
614                         lenght = MAX_BUFFERED_BYTES;
615                     }
616 
617                     MultipartRequestEntity mre = AsyncHttpProviderUtils.createMultipartRequestEntity(request.getParts(), request.getHeaders());
618 
619                     urlConnection.setRequestProperty("Content-Type", mre.getContentType());
620                     urlConnection.setRequestProperty("Content-Length", String.valueOf(mre.getContentLength()));
621 
622                     mre.writeRequest(urlConnection.getOutputStream());
623                 } else if (request.getEntityWriter() != null) {
624                     int lenght = (int) request.getContentLength();
625                     if (lenght != -1) {
626                         urlConnection.setRequestProperty("Content-Length", String.valueOf(lenght));
627                         urlConnection.setFixedLengthStreamingMode(lenght);
628                     }
629                     request.getEntityWriter().writeEntity(urlConnection.getOutputStream());
630                 } else if (request.getFile() != null) {
631                     File file = request.getFile();
632                     if (!file.isFile()) {
633                         throw new IOException(String.format(Thread.currentThread()
634                                 + "File %s is not a file or doesn't exist", file.getAbsolutePath()));
635                     }
636                     urlConnection.setRequestProperty("Content-Length", String.valueOf(file.length()));
