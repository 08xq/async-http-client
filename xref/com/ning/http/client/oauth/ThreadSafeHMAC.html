17  package com.ning.http.client.oauth;
18  
19  import com.ning.http.util.UTF8Codec;
20  import com.ning.http.util.UTF8UrlEncoder;
21  
22  import javax.crypto.Mac;
23  import javax.crypto.spec.SecretKeySpec;
24  
25  /**
26   * Since cloning (of MAC instances)  is not necessarily supported on all platforms
27   * (and specifically seems to fail on MacOS), let's wrap synchronization/reuse details here.
28   * Assumption is that this is bit more efficient (even considering synchronization)
29   * than locating and reconstructing instance each time.
30   * In future we may want to use soft references and thread local instance.
31   *
32   * @author tatu (tatu.saloranta@iki.fi)
33   */
34  public class ThreadSafeHMAC {
35      private static final String HMAC_SHA1_ALGORITHM = "HmacSHA1";
36  
37      private final Mac mac;
38  
39      public ThreadSafeHMAC(ConsumerKey consumerAuth, RequestToken userAuth) {
40          byte[] keyBytes = UTF8Codec.toUTF8(UTF8UrlEncoder.encode(consumerAuth.getSecret()) + "&" + UTF8UrlEncoder.encode(userAuth.getSecret()));
41          SecretKeySpec signingKey = new SecretKeySpec(keyBytes, HMAC_SHA1_ALGORITHM);
42  
43          // Get an hmac_sha1 instance and initialize with the signing key
44          try {
45              mac = Mac.getInstance(HMAC_SHA1_ALGORITHM);
46              mac.init(signingKey);
47          } catch (Exception e) {
48              throw new IllegalArgumentException(e);
49          }
50  
51      }
52  
53      public synchronized byte[] digest(byte[] message) {
54          mac.reset();
55          return mac.doFinal(message);
56      }
57  }
