38       * @param offset the offset where to start parsing
39       * @param length the number of chars to parse
40       */
41      public RFC2616DateParser(String string, int offset, int length) {
42  
43          if (string.length() + offset < length)
44              throw new IllegalArgumentException("String length doesn't match offset and length");
45  
46          this.string = string;
47          this.offset = offset;
48          this.length = length;
49      }

51      private static class Tokens {
52          public final int[] starts;
53          public final int[] ends;
54          public final int length;
55  
56          public Tokens(int[] starts, int[] ends, int length) {
57              this.starts = starts;
58              this.ends = ends;
59              this.length = length;
60          }
61      }

63      private Tokens tokenize() {
64  
65          int[] starts = new int[8];
66          int[] ends = new int[8];
67          boolean inToken = false;
68          int tokenCount = 0;
69  
70          int end = offset + length;
71          for (int i = offset; i < end; i++) {
72  
73              char c = string.charAt(i);
74              if (c == ' ' || c == ',' || c == '-' || c == ':') {
75                  if (inToken) {
76                      ends[tokenCount++] = i;
77                      inToken = false;
78                  }
79              } else if (!inToken) {
80                  starts[tokenCount] = i;
81                  inToken = true;
82              }
83          }
84  
85          // finish lastToken
86          if (inToken)
87              ends[tokenCount++] = end;
88  
89          return new Tokens(starts, ends, tokenCount);
90      }

92      /**
93       * @param validate if validation is to be enabled of non-critical elements,
94       *            such as day of week and timezone
95       * @return null is the string is not a valid RFC2616 date
96       */
97      public RFC2616Date parse() {
98  
99          Tokens tokens = tokenize();
100 
101         if (tokens.length != 7 && tokens.length != 8)
102             return null;
103 
104         // 1st token is ignored: ignore day of week
105         // 8th token is ignored: supposed to always be GMT
106 
107         if (isDigit(string.charAt(tokens.starts[1])))
108             return buildDate(tokens);
109         else
110             return buildANSICDate(tokens);
111     }

113     private RFC2616Date buildDate(Tokens tokens) {
114 
115         // Sun, 06 Nov 1994 08:49:37 GMT
116 
117         Builder dateBuilder = new Builder();
118 
119         if (isValidDayOfMonth(tokens.starts[1], tokens.ends[1], dateBuilder) && //
120                 isValidMonth(tokens.starts[2], tokens.ends[2], dateBuilder) && //
121                 isValidYear(tokens.starts[3], tokens.ends[3], dateBuilder) && //
122                 isValidHour(tokens.starts[4], tokens.ends[4], dateBuilder) && //
123                 isValidMinuteSecond(tokens.starts[5], tokens.ends[5], dateBuilder, true) && //
124                 isValidMinuteSecond(tokens.starts[6], tokens.ends[6], dateBuilder, false)) {
125             return dateBuilder.build();
126         }
127 
128         return null;
129     }

131     private RFC2616Date buildANSICDate(Tokens tokens) {
132 
133         // Sun Nov 6 08:49:37 1994
134 
135         Builder dateBuilder = new Builder();

137         if (isValidMonth(tokens.starts[1], tokens.ends[1], dateBuilder) && //

142                 isValidYear(tokens.starts[6], tokens.ends[6], dateBuilder)) {
143             return dateBuilder.build();
144         }
145 
146         return null;
147     }

170     private boolean isValidDayOfMonth(int start, int end, Builder dateBuilder) {
171 
172         int tokenLength = end - start;
173 
174         if (tokenLength == 1) {
175             char c0 = string.charAt(start);
176             return isValid1DigitDayOfMonth(c0, dateBuilder);

178         } else if (tokenLength == 2) {

202         return false;

273     private boolean isValidMonth(int start, int end, Builder dateBuilder) {
274 
275         if (end - start != 3)
276             return false;
277 
278         char c0 = string.charAt(start);
279         char c1 = string.charAt(start + 1);
280         char c2 = string.charAt(start + 2);
281 
282         return isValidJanuaryJuneJuly(c0, c1, c2, dateBuilder) || //
283                 isValidFebruary(c0, c1, c2, dateBuilder) || //
284                 isValidMarchMay(c0, c1, c2, dateBuilder) || //
285                 isValidAprilAugust(c0, c1, c2, dateBuilder) || //
286                 isValidSeptember(c0, c1, c2, dateBuilder) || //
287                 isValidOctober(c0, c1, c2, dateBuilder) || //
288                 isValidNovember(c0, c1, c2, dateBuilder) || //
289                 isValidDecember(c0, c1, c2, dateBuilder);
290     }

292     private boolean isValid2DigitsYear(char c0, char c1, Builder dateBuilder) {
293         if (isDigit(c0) && isDigit(c1)) {
294             int i0 = getNumericValue(c0);
295             int i1 = getNumericValue(c1);
296             int year = i0 * 10 + i1;
297             year = year < 70 ? year + 2000 : year + 1900;
298 
299             return setValidYear(year, dateBuilder);
300         }
301         return false;
302     }

326 
327         int length = end - start;

329         if (length == 2) {

331             char c1 = string.charAt(start + 1);

338             char c3 = string.charAt(start + 3);
339             return isValid4DigitsYear(c0, c1, c2, c3, dateBuilder);
340         }
