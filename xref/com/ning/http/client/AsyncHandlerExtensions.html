13  package com.ning.http.client;
14  
15  /**
16   * This interface hosts new low level callback methods on {@link AsyncHandler}.
17   * For now, those methods are in a dedicated interface in order not to break the existing API,
18   * but could be merged into one of the existing ones in AHC 2.
19   * 
20   * More additional hooks might come, such as:
21   * <ul>
22   *   <li>onConnected()</li>
23   *   <li>onConnectionClosed()</li>
24   *   <li>onBytesSent(long numberOfBytes)</li>
25   *   <li>onBytesReceived(long numberOfBytes)</li>
26   * </ul>
27   */
28  public interface AsyncHandlerExtensions {
29  
30      /**
31       * Notify the callback when a request is being written on the wire.
32       * If the original request causes multiple requests to be sent, for example, because of authorization or retry,
33       * it will be notified multiple times.
34       * Currently only supported by the Netty provider.
35       */
36      void onRequestSent();
37  
38      /**
39       * Notify the callback every time a request is being retried.
40       */
41      void onRetry();
42  }
