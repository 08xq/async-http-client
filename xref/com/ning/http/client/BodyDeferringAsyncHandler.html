26  /**
27   * An AsyncHandler that returns Response (without body, so status code and
28   * headers only) as fast as possible for inspection, but leaves you the option
29   * to defer body consumption.
30   * <p/>
31   * This class introduces new call: getResponse(), that blocks caller thread as
32   * long as headers are received, and return Response as soon as possible, but
33   * still pouring response body into supplied output stream. This handler is
34   * meant for situations when the "recommended" way (using
35   * <code>client.prepareGet("http://foo.com/aResource").execute().get()</code>
36   * would not work for you, since a potentially large response body is about to
37   * be GETted, but you need headers first, or you don't know yet (depending on
38   * some logic, maybe coming from headers) where to save the body, or you just
39   * want to leave body stream to some other component to consume it.
40   * <p/>
41   * All these above means that this AsyncHandler needs a bit of different
42   * handling than "recommended" way. Some examples:
43   * <p/>
44   * <pre>
45   *     FileOutputStream fos = ...
46   *     BodyDeferringAsyncHandler bdah = new BodyDeferringAsyncHandler(fos);
47   *     // client executes async
48   *     Future&lt;Response&gt; fr = client.prepareGet(&quot;http://foo.com/aresource&quot;).execute(
49   * 	bdah);
50   *     // main thread will block here until headers are available
51   *     Response response = bdah.getResponse();
52   *     // you can continue examine headers while actual body download happens
53   *     // in separate thread
54   *     // ...
55   *     // finally &quot;join&quot; the download
56   *     fr.get();
57   * </pre>
58   * <p/>
59   * <pre>
60   *     PipedOutputStream pout = new PipedOutputStream();
61   *     BodyDeferringAsyncHandler bdah = new BodyDeferringAsyncHandler(pout);
62   *     // client executes async
63   *     Future&lt;Response&gt; fr = client.prepareGet(&quot;http://foo.com/aresource&quot;).execute(bdah);
64   *     // main thread will block here until headers are available
65   *     Response response = bdah.getResponse();
66   *     if (response.getStatusCode() == 200) {
67   *      InputStream pin = new BodyDeferringInputStream(fr,new PipedInputStream(pout));
68   *      // consume InputStream
69   *      ...
70   *     } else {
71   *      // handle unexpected response status code
72   *      ...
73   *     }
74   * </pre>
75   */
76  public class BodyDeferringAsyncHandler implements AsyncHandler<Response> {
77      private final ResponseBuilder responseBuilder = new ResponseBuilder();
78  
79      private final CountDownLatch headersArrived = new CountDownLatch(1);
80  
81      private final OutputStream output;
82  
83      private boolean responseSet;
84  
85      private volatile Response response;
86  
87      private volatile Throwable throwable;
88  
89      private final Semaphore semaphore = new Semaphore(1);
90  
91      public BodyDeferringAsyncHandler(final OutputStream os) {
92          this.output = os;
93          this.responseSet = false;
94      }
95  
96      public void onThrowable(Throwable t) {
97          this.throwable = t;
98          // Counting down to handle error cases too.
99          // In "premature exceptions" cases, the onBodyPartReceived() and
100         // onCompleted()
101         // methods will never be invoked, leaving caller of getResponse() method
102         // blocked forever.
103         try {
104             semaphore.acquire();
105         } catch (InterruptedException e) {
106             // Ignore
107         } finally {
108             headersArrived.countDown();
109             semaphore.release();
110         }
111 
112         try {
113             closeOut();
114         } catch (IOException e) {
115             // ignore
116         }
117     }
118 
119     public STATE onStatusReceived(HttpResponseStatus responseStatus)
120             throws Exception {
121         responseBuilder.reset();
122         responseBuilder.accumulate(responseStatus);
123         return STATE.CONTINUE;
124     }
125 
126     public STATE onHeadersReceived(HttpResponseHeaders headers)
127             throws Exception {
128         responseBuilder.accumulate(headers);
129         return STATE.CONTINUE;
130     }
131 
132     public STATE onBodyPartReceived(HttpResponseBodyPart bodyPart)
133             throws Exception {
134         // body arrived, flush headers
135         if (!responseSet) {
136             response = responseBuilder.build();
137             responseSet = true;
138             headersArrived.countDown();
139         }
140 
141         bodyPart.writeTo(output);
142         return STATE.CONTINUE;
143     }
144 
145     protected void closeOut() throws IOException {
146         try {
147             output.flush();
148         } finally {
149             output.close();
150         }
151     }
152 
153     public Response onCompleted() throws IOException {
154 
155         if (!responseSet) {
156             response = responseBuilder.build();
157             responseSet = true;
158         }
159 
160         // Counting down to handle error cases too.
161         // In "normal" cases, latch is already at 0 here
162         // But in other cases, for example when because of some error
163         // onBodyPartReceived() is never called, the caller
164         // of getResponse() would remain blocked infinitely.
165         // By contract, onCompleted() is always invoked, even in case of errors
166         headersArrived.countDown();
167 
168         closeOut();
169 
170         try {
171             semaphore.acquire();
172             if (throwable != null) {
173                 IOException ioe = new IOException(throwable.getMessage());
174                 ioe.initCause(throwable);
175                 throw ioe;
176             } else {
177                 // sending out current response
178                 return responseBuilder.build();
179             }
180         } catch (InterruptedException e) {
181             return null;
182         } finally {
183             semaphore.release();
184         }
185     }
186 
187     /**
188      * This method -- unlike Future<Reponse>.get() -- will block only as long,
189      * as headers arrive. This is useful for large transfers, to examine headers
190      * ASAP, and defer body streaming to it's fine destination and prevent
191      * unneeded bandwidth consumption. The response here will contain the very
192      * 1st response from server, so status code and headers, but it might be
193      * incomplete in case of broken servers sending trailing headers. In that
194      * case, the "usual" Future<Response>.get() method will return complete
195      * headers, but multiple invocations of getResponse() will always return the
196      * 1st cached, probably incomplete one. Note: the response returned by this
197      * method will contain everything <em>except</em> the response body itself,
198      * so invoking any method like Response.getResponseBodyXXX() will result in
199      * error! Also, please not that this method might return <code>null</code>
200      * in case of some errors.
201      *
202      * @return a {@link Response}
203      * @throws InterruptedException
204      */
205     public Response getResponse() throws InterruptedException, IOException {
206         // block here as long as headers arrive
207         headersArrived.await();
208 
209         try {
210             semaphore.acquire();
211             if (throwable != null) {
212                 IOException ioe = new IOException(throwable.getMessage());
213                 ioe.initCause(throwable);
214                 throw ioe;
215             } else {
216                 return response;
217             }
218         } finally {
219             semaphore.release();
220         }
221     }

225     /**
226      * A simple helper class that is used to perform automatic "join" for async
227      * download and the error checking of the Future of the request.
228      */
229     public static class BodyDeferringInputStream extends FilterInputStream {
230         private final Future<Response> future;
231 
232         private final BodyDeferringAsyncHandler bdah;
233 
234         public BodyDeferringInputStream(final Future<Response> future,
235                                         final BodyDeferringAsyncHandler bdah, final InputStream in) {
236             super(in);
237             this.future = future;
238             this.bdah = bdah;
239         }
240 
241         /**
242          * Closes the input stream, and "joins" (wait for complete execution
243          * together with potential exception thrown) of the async request.
244          */
245         public void close() throws IOException {
246             // close
247             super.close();
248             // "join" async request
249             try {
250                 getLastResponse();
251             } catch (Exception e) {
252                 IOException ioe = new IOException(e.getMessage());
253                 ioe.initCause(e);
254                 throw ioe;
255             }
256         }
257 
258         /**
259          * Delegates to {@link BodyDeferringAsyncHandler#getResponse()}. Will
260          * blocks as long as headers arrives only. Might return
261          * <code>null</code>. See
262          * {@link BodyDeferringAsyncHandler#getResponse()} method for details.
263          *
264          * @return a {@link Response}
265          * @throws InterruptedException
266          */
267         public Response getAsapResponse() throws InterruptedException,
268                 IOException {
269             return bdah.getResponse();
270         }
271 
272         /**
273          * Delegates to <code>Future<Response>#get()</code> method. Will block
274          * as long as complete response arrives.
275          *
276          * @return a {@link Response}
277          * @throws InterruptedException
278          * @throws ExecutionException
279          */
280         public Response getLastResponse() throws InterruptedException,
281                 ExecutionException {
282             return future.get();
283         }
284     }
